# Accordion

Divide content into collapsible sections.

### MpAccordion

#### MpAccordion Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default-index-open` | `-` | `-` | Default index for opened accordion. |
| `is-allow-multiple` | `-` | `-` | If true, multiple accordion items can be expanded at once. |
| `is-allow-toggle` | `-` | `-` | If true, any expanded accordion item can be collapsed again. |
| `is-controlled` | `-` | `-` | If true, accordion will be controll manually. |

#### MpAccordion Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `change` | `-` | `-` | Function for handling change event. |

#### MpAccordion Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Accordion slot |

### MpAccordionItem

#### MpAccordionItem Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of accordion item. |
| `is-open` | `-` | `-` | If true, accordion item will be open. |
| `is-default-open` | `-` | `-` | If true, accordion item will be initially opened. |
| `is-disabled` | `-` | `-` | If true, accordion item will be disabled. |

#### MpAccordionItem Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Accordion item slot |

### MpAccordionHeader

#### MpAccordionHeader Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `is-clickable` | `-` | `-` | If true, accordion header is clickable |

#### MpAccordionHeader Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Accordion header slot |
| `description` | `-` | `-` | Accordion header description slot |

### MpAccordionIcon

#### MpAccordionIcon Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `-` | `-` | Name of accordion icon. |

### MpAccordionPanel

#### MpAccordionPanel Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Accordion panel slot |

### Usage: Controlled

```vue
<script setup lang="ts">
  import { ref } from 'vue'
  import {
    MpAccordion,
    MpAccordionHeader,
    MpAccordionIcon,
    MpAccordionItem,
    MpAccordionPanel,
    MpText,
    css
  } from '@mekari/pixel3'

  const headerTextClass = css({
    flex: '1',
    textAlign: 'left',
    fontWeight: 'semiBold'
  })

  const accordionItems = ref([
    {
      id: 1,
      isOpen: false,
      content: {
        title: 'Integration between Mekari products',
        description:
          'Manage various business operations more optimally with Mekari product ecosystem that is integrated with each other.'
      }
    },
    {
      id: 2,
      isOpen: false,
      content: {
        title: 'Integration between Mekari products',
        description:
          'Manage various business operations more optimally with Mekari product ecosystem that is integrated with each other.'
      }
    },
    {
      id: 3,
      isOpen: false,
      content: {
        title: 'Integration between Mekari products',
        description:
          'Manage various business operations more optimally with Mekari product ecosystem that is integrated with each other.'
      }
    }
  ])

  function handleClick(index: number) {
    accordionItems.value[index].isOpen = !accordionItems.value[index].isOpen
  }
</script>

<template>
  <div>
    <MpAccordion is-controlled id="accordion-controlled">
      <MpAccordionItem
        v-for="(item, index) in accordionItems"
        :key="index"
        :is-open="item.isOpen"
        @click="handleClick(index)"
        :id="`accordion-item-${index}`"
      >
        <MpAccordionHeader>
          <MpText size="h2" :class="headerTextClass">
            {{ item.content.title }}
          </MpText>

          <MpAccordionIcon />
        </MpAccordionHeader>

        <MpAccordionPanel>
          <MpText color="inherit">
            {{ item.content.description }}
          </MpText>
        </MpAccordionPanel>
      </MpAccordionItem>
    </MpAccordion>
  </div>
</template>

```

# Airene Chat Input

Chat Input component for Airene.

### MpAireneChatInput

#### MpAireneChatInput Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of chat input. |
| `model-value` | `-` | `-` | modelValue of chat input, use for v-model. |
| `placeholder` | `-` | `-` | Placeholder of chat input. |
| `maxlength` | `-` | `-` | Max length of chat input. |
| `aria-labelled` | `-` | `-` | Optional prop for accessibility. |
| `aria-labelled-by` | `-` | `-` | Optional prop for accessibility. |
| `is-read-only` | `-` | `-` | If true, chat input will be read only. |
| `is-disabled` | `-` | `-` | If true, chat input will be disabled. |
| `is-invalid` | `-` | `-` | If true, chat input will be invalid. |
| `is-required` | `-` | `-` | If true, chat input will be required. |
| `is-force-focus` | `-` | `-` | If true, chat input will be force to focus state |

#### MpAireneChatInput Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `change` | `-` | `-` | Function for handling change event. |
| `input` | `-` | `-` | Function for handling input event. |
| `focus` | `-` | `-` | Function for handling focus event. |
| `blur` | `-` | `-` | Function for handling blur event. |
| `keydown` | `-` | `-` | Function for handling keyboard event. |
| `send` | `-` | `-` | Function for handling send event. |
| `update:modelValue` | `-` | `-` | Function for handling v-model. |

#### MpAireneChatInput Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Slot to render child element. |

### Usage: Basic

```vue
<template>
  <MpFlex direction="column" gap="1.5">
    <MpAireneChatInput
      id="basic"
      v-model="chat"
      @focus="handleFocus"
      @blur="handleBlur"
      @send="handleSendMessage"
      @keydown="handleKeydown"
    />
    <MpText size="label-small" :color="isNextTheme ? 'text.secondary' : 'gray.400'">
      Airene responses can be inaccurate or misleading.
      <MpText
        as="span"
        size="label-small"
        :color="isNextTheme ? 'text.link' : 'blue.400'"
        is-text-link
      >
        Learn more
      </MpText>
    </MpText>
  </MpFlex>
</template>

<script setup lang="ts">
  import { ref } from 'vue'
  import { MpFlex, MpAireneChatInput, MpText } from '@mekari/pixel3'
  import { usePixelTheme } from '@mekari/pixel3-utils'

  const { isNextTheme } = usePixelTheme()
  const chat = ref('')

  function handleFocus() {
    console.log('Airene chat input focused')
  }

  function handleBlur() {
    console.log('Airene chat input blurred')
  }

  function handleSendMessage() {
    console.log('Message sent:', chat.value)
    chat.value = ''
  }

  function handleKeydown(e: KeyboardEvent) {
```

```vue
<script setup lang="ts">
  import { ref } from 'vue'
  import { css, MpFlex, MpAireneChatInput, MpText, MpImage, MpIcon } from '@mekari/pixel3'
  import { usePixelTheme } from '@mekari/pixel3-utils'

  const { isNextTheme } = usePixelTheme()
  const chat = ref('')
  const fileInput = ref()
  const previews = ref<string[]>([])

  function triggerUpload() {
    fileInput.value.click()
  }

  function handleFiles(event: Event) {
    const input = event.target as HTMLInputElement
    if (!input.files) return

    const files = Array.from(input.files)
    previews.value = []

    files.forEach((file) => {
      const reader = new FileReader()

      reader.onload = (e: ProgressEvent<FileReader>) => {
        const result = e.target?.result
        if (typeof result === 'string') {
          previews.value.push(result)
        }
      }

      reader.readAsDataURL(file)
    })

    input.value = ''
  }

  function removeImage(index: number) {
    previews.value.splice(index, 1)
  }
```

# Avatar

Used to display profile image, initial or fallback icon.

### MpAvatar

#### MpAvatar Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of avatar. |
| `name` | `-` | `-` | Name of avatar. |
| `variant` | `-` | `-` | Variant of avatar. |
| `variant-color` | `-` | `-` | Variant color of avatar. Not set will be get random color. |
| `size` | `-` | `-` | Size of avatar. |
| `has-border` | `-` | `-` | If true, avatar border will be displayed. |
| `border-color` | `-` | `-` | Border color of avatar. |
| `src` | `-` | `-` | Image url of avatar. |
| `icon` | `-` | `-` | Icon of avatar. |
| `icon-variant` | `-` | `-` | Icon variant of avatar. |
| `icon-color` | `-` | `-` | Icon color of avatar. |
| `spacing` | `-` | `-` | Spacing of avatar. |

#### MpAvatar Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `click` | `-` | `-` | Function for handling click event. |
| `load` | `-` | `-` | Function for handling load image. |
| `error` | `-` | `-` | Function for handling error image. |

# Chart

Chart visualizes data in an application. The type of chart you use will depend on your use case and the type of data you need to display.

### MpChart

#### MpChart Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of chart. |
| `title` | `-` | `-` | Title of chart. |
| `type` | `-` | `-` | Type of chart. |
| `width-chart` | `-` | `-` | Width of chart. |
| `height-chart` | `-` | `-` | Height of chart. |
| `width-container` | `-` | `-` | Width of chart container. |
| `height-container` | `-` | `-` | Height of chart container. |
| `data` | `-` | `-` | Data of chart. |
| `options` | `-` | `-` | Options of chart. |
| `color-pattern` | `-` | `-` | Pattern color of chart. |
| `color-start` | `-` | `-` | Starting number for color of chart. from 1 to 16. |
| `color-ratio` | `-` | `-` | Ratio shades for color of chart. from 0.1 to 0.9, only apply when color pattern is dark-shades or light-shades. |
| `legend-position` | `-` | `-` | Position of legend from chart. |
| `legend-direction` | `-` | `-` | Direction of chart legend. |
| `tooltip-margin-top` | `-` | `-` | Margin top value for chart tooltip, if use custom legend. |
| `tooltip-margin-left` | `-` | `-` | Margin left value for chart tooltip, if use custom legend. |
| `tooltip-options` | `-` | `-` | Options for tooltip of chart. |
| `is-show-legend` | `-` | `-` | If true, custom chart legend will showed. |
| `is-show-tooltip` | `-` | `-` | If true, custom chart tooltip will showed. |
| `is-show-data-labels` | `-` | `-` | If true, custom chart legend will showed. |
| `is-horizontal` | `-` | `-` | If true, will render horizontal chart. |
| `is-stacked` | `-` | `-` | If true, will render stacked chart. |
| `is-area` | `-` | `-` | If true, will render area chart. |
| `is-debug` | `-` | `-` | If true, debug info will showed in console. |
| `is-show-background-hover` | `-` | `-` | If true, background color changes when hovered. |
| `is-show-gauge-pointer` | `-` | `-` | If true, show gauge pointer on chart. |

#### MpChart Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `show-tooltip` | `-` | `-` | Function for handling show tooltip. |
| `hide-tooltip` | `-` | `-` | Function for handling hide tooltip. |
| `click-legend` | `-` | `-` | Function for handling click legend. |

#### MpChart Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Slot to render elements inside chart. |
| `header` | `-` | `-` | Slot to render elements for header chart. |
| `legend` | `-` | `-` | Slot to render elements for custom legend. |
| `action` | `-` | `-` | Slot to render elements for action button. |

# Checkbox

Allows user to toggle between checked, unchecked, and indeterminate states.

### MpCheckbox

#### MpCheckbox Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of checkbox. |
| `name` | `-` | `-` | Name of checkbox. |
| `value` | `-` | `-` | Value of checkbox. |
| `model-value` | `-` | `-` | modelValue for group of checkbox. |
| `is-checked` | `-` | `-` | If true, checkbox will be checked. |
| `is-indeterminate` | `-` | `-` | If true, checkbox will be indetermintate. |
| `is-disabled` | `-` | `-` | If true, checkbox will be disable. |
| `is-invalid` | `-` | `-` | If true, checkbox will be invalid. |
| `is-read-only` | `-` | `-` | If true, checkbox will be read only. |

#### MpCheckbox Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `click` | `-` | `-` | Function for handling click event. |
| `change` | `-` | `-` | Function for handling change event. |
| `update:model-value` | `-` | `-` | Function for update modelValue prop. |
| `update:is-checked` | `-` | `-` | Function for update modelValue prop. |

#### MpCheckbox Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Slot to render checkbox label. |
| `description` | `-` | `-` | Slot to render checkbox description. |

### Usage: Checkbox Group

```vue
<template>
  <MpFlex gap="4" mb="4">
    <MpCheckbox
      id="checkbox-10"
      :is-indeterminate="isIndeterminate"
      :is-checked="checkedNames.length === 2"
      @change="handleCheckAll"
      >Check All</MpCheckbox
    >
    <MpCheckbox id="checkbox-11" value="check_one" v-model="checkedNames" is-disabled
      >Check One</MpCheckbox
    >
    <MpCheckbox id="checkbox-12" value="check_two" v-model="checkedNames">Check Two</MpCheckbox>
    <MpCheckbox id="checkbox-13" value="check_three" v-model="checkedNames">Check Three</MpCheckbox>
  </MpFlex>

  <MpText>Value: {{ checkedNames }}</MpText>
</template>

<script setup lang="ts">
  import { ref, computed } from 'vue'
  import { MpCheckbox, MpText, MpFlex } from '@mekari/pixel3'

  let checkedNames = ref([] as string[])

  const isIndeterminate = computed(
    () => checkedNames.value?.length > 0 && checkedNames.value?.length < 2
  )

  function handleCheckAll(val: boolean): void {
    if (val) {
      checkedNames.value = ['check_two', 'check_three']
    } else {
      checkedNames.value = []
    }
  }
</script>

```

### Usage: With Description

```vue
<template>
  <MpCheckbox id="checkbox-2" :is-checked="isChecked" @change="handleChange" @click="handleClick"
    >Marketing promotion
    <template #description>
      Send promotional offers, product announcements, and more to increase awareness and engagement.
    </template>
  </MpCheckbox>
</template>

<script setup lang="ts">
  import { ref } from 'vue'
  import { MpCheckbox } from '@mekari/pixel3'

  let isChecked = ref(false)

  function handleChange(val: boolean): void {
    isChecked.value = val
  }

  function handleClick(): void {
    alert('Receive marketing promos')
  }
</script>

```

### Usage: With Area Upload

```vue
<template>
  <div :class="css({ width: 'full', minHeight: '500px', height: 'full', overflow: 'auto' })">
    <MpDropzone
      overlay-variant="black"
      placeholder="Drop your file(s) here"
      :has-custom-upload="true"
      :is-loading="isLoading"
      @dragover="(e) => log('dragover', e)"
      @dragleave="(e) => log('dragleave', e)"
      @drop="(e) => log('drop', e)"
      @change="handleChange"
    >
      <MpText size="h2">Drag Image(s) to This Area</MpText>
      <div v-for="(item, index) in images" :key="index" :class="css({ width: '400px' })">
        <img :src="handlePreview(item)" />
      </div>
    </MpDropzone>
  </div>
</template>

<script setup lang="ts">
  import { ref } from 'vue'
  import { MpDropzone, MpText, css } from '@mekari/pixel3'

  const isLoading = ref(false)
  const images = ref()

  function log(message: string, e: Event) {
    console.log(message, e)
  }

  function handleChange(item: File) {
    isLoading.value = true
    console.log('IMAGE: ', item)
    images.value = item
    setTimeout(() => {
      // Upload image to server
      isLoading.value = false
    }, 2000)
  }

  function handlePreview(item: File) {
    const REGEXP_MIME_TYPE_IMAGES = /^image\/\w+$/
    if (REGEXP_MIME_TYPE_IMAGES.test(item.type)) {
      return URL.createObjectURL(item)
    }
  }
</script>

```

# Popover

Popovers are used to provide supplemental, useful, unique information about an element, including one or multiple actionable elements. They inform the user of additional information within the context of their original view, but without forcing the user to act upon it like a modal.

### MpPopover

#### MpPopover Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `placement` | `-` | `-` | Placement of popover. |
| `id` | `-` | `-` | Id of popover. |
| `trigger` | `-` | `-` | Trigger of popover. |
| `trigger-id` | `-` | `-` | Trigger Id of popover. |
| `is-default-open` | `-` | `-` | If true, popover will open in first render. |
| `is-manual` | `-` | `-` | Popover show/hide will be handle manual with `isOpen` prop. |
| `is-open` | `-` | `-` | If true/false, popover will show/hide. Only active if `isManual` prop is true. |
| `is-focus-on-close` | `-` | `-` | If true, focus will be returned to the element that triggers the popover when it closes. |
| `is-close-on-blur` | `-` | `-` | If true, the popover will close when you blur out it by clicking outside or tabbing out. |
| `is-close-on-escape` | `-` | `-` | If true, close the popover when the esc key is pressed. |
| `is-close-on-select` | `-` | `-` | If true, close the popover when the select item from Popover List Item. |
| `use-portal` | `-` | `-` | If true, popover content is rendered with portal. |
| `is-keep-alive` | `-` | `-` | If true, popover will be hidden using CSS instead of destroying DOM. |
| `is-adaptive-width` | `-` | `-` | If true, popover content width will have min-width same as popover trigger. |

```vue
  useInfiniteScroll(
    el,
    () => {
      if (!isLoading.value) {
        isLoading.value = true

        setTimeout(() => {
          const length = data.value.length + 1
          data.value.push(...Array.from({ length: 10 }, (_, i) => length + i))

          isLoading.value = false
        }, 300)
      }
    },
    { distance: 10 }
  )

  function handlePopoverOpen() {
    const popoverContentElement = popoverContentNode.value.$el.nextElementSibling
    el.value = popoverContentElement
  }
```

```vue
      :value="0.5"
      :min="0"

      :step="0.1"

      :is-gradient="true"
      @input="handleInput"
      @change="handleChange"
    >

        {{ value }}
      </template>
      <template #min> Reserved </template>
      <template #max> Creative </template>

  const changeVal = ref(0)
  const inputVal = ref('')

  function handleChange(val: number) {
    changeVal.value = val
  }

  function handleInput(val: string) {
    inputVal.value = val
  }
</script>
```

# Toast

### toast

#### toast Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of toast. |
| `title` | `-` | `-` | Title of toast. |
| `variant` | `-` | `-` | Variant of toast. |
| `duration` | `-` | `-` | Duration of toast. |
| `position` | `-` | `-` | Position of toast. |
| `render` | `-` | `-` | Render VNode to custom content instead of title prop. |

### Usage: Variant

```vue
<script setup lang="ts">
  import { toast, MpFlex, MpButton } from '@mekari/pixel3'

  function handleShowSuccess() {
    toast.notify({
      id: 'toast-wtm',
      position: 'top-center',
      variant: 'success',
      title: 'Setting is imported.'
    })
  }

  function handleShowError() {
    toast.notify({
      id: 'toast-0tn',
      position: 'top-center',
      variant: 'error',
      title: 'Server error, please try again.'
    })
  }

  function handleShowGreeting() {
    toast.notify({
      id: 'toast-h7d',
      position: 'top-center',
      variant: 'greeting',
      title: '🌤️ Good morning, Fajar'
    })
  }

  function handleClear() {
    toast.closeAll()
  }
</script>

<template>
  <MpFlex gap="2">
    <MpButton variant="primary" data-testid="show-toast-wtm" @click="handleShowSuccess"
      >Show success</MpButton
    >
    <MpButton variant="danger" data-testid="show-toast-0tn" @click="handleShowError"
      >Show error</MpButton
    >
    <MpButton variant="ghost" data-testid="show-toast-h7d" @click="handleShowGreeting"
      >Show greeting</MpButton
    >
    <MpButton variant="secondary" @click="handleClear">Clear all</MpButton>
  </MpFlex>
</template>

```

# Toggle

Alternate appearance for a toggle that simulates on/off instead of checked/unchecked.

### MpToggle

#### MpToggle Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of toggle. |
| `name` | `-` | `-` | Name of toggle. |
| `value` | `-` | `-` | Value of toggle. |
| `is-checked` | `-` | `-` | If true, toggle will be checked. |
| `is-disabled` | `-` | `-` | If true, toggle will be disable. |
| `is-invalid` | `-` | `-` | If true, toggle will be invalid. |
| `is-read-only` | `-` | `-` | If true, toggle will be read only. |

#### MpToggle Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `click` | `-` | `-` | Function for handling click event. |
| `change` | `-` | `-` | Function for handling change event. |
| `update:is-checked` | `-` | `-` | Function for update modelValue prop. |

#### MpToggle Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Slot to render toggle label. |
| `description` | `-` | `-` | Slot to render toggle description. |

# Tooltip

A tooltip is a brief, informative message that appears when a user interacts with an element.

### MpTooltip

#### MpTooltip Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `-` | `-` | Label of tooltip. |
| `id` | `-` | `-` | Id of tooltip. |
| `placement` | `-` | `-` | Placement of tooltip. |
| `is-default-open` | `-` | `-` | If true, tooltip will be open by default. |
| `is-manual` | `-` | `-` | If true, tooltip show/hide will be handle manual with `isOpen` prop. |
| `is-open` | `-` | `-` | If true/false, tooltip will show/hide. Only active if `is-manual` prop is true. |
| `show-delay` | `-` | `-` | Show delay of tooltip in millisecond. |
| `hide-delay` | `-` | `-` | Hide delay of tooltip in millisecond. |
| `use-portal` | `-` | `-` | If true, tooltip is rendered with portal. |
| `is-keep-alive` | `-` | `-` | If true, tooltip will be hidden using CSS instead of destroying DOM. |

#### MpTooltip Events

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `-` | `-` | Function for handling open event. |
| `close` | `-` | `-` | Function for handling close event. |

#### MpTooltip Slots

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `default` | `-` | `-` | Tooltip slot |

# Upload

Upload files through a file input form element.

### MpUpload

#### MpUpload Props

| Name | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `-` | `-` | Id of upload. |
| `file-list` | `-` | `-` | List of files for input. |
| `accept` | `-` | `-` | List of file types for input. |
| `name` | `-` | `-` | Name for input. |
| `placeholder` | `-` | `-` | Placeholder for input. |
| `loading-text` | `-` | `-` | Text in loading state. |
| `button-text` | `-` | `-` | Text in button upload. |
| `is-loading` | `-` | `-` | If true, show loading state. |
| `is-disabled` | `-` | `-` | If true, upload will be disable. |
| `is-invalid` | `-` | `-` | If true, upload will be invalid. |
| `is-required` | `-` | `-` | If true, mark as required. |
| `is-reset-on-change` | `-` | `-` | If true, input value will reset when file change. |
| `is-multiple` | `-` | `-` | If true, input receive multiple files to select. |
| `is-full-width` | `-` | `-` | If true, upload will have full width. |

### Usage: With Upload List

```vue
<template>
  <div>
    <MpUpload
      accept=".jpg, .jpeg, .png, .webp"
      :fileList="images"
      :is-multiple="true"
      :is-reset-on-change="false"
      :is-loading="isLoading"
      :is-invalid="isInvalid"
      @change="handleChange"
      @clear="handleClear"
    />
    <MpFlex direction="column" w="65" mt="5">
      <MpUploadList
        v-for="(item, index) in images"
        :id="`upload-list-${index}`"
        :key="index"
        :title="item.name"
        :status="handleStatus(item)"
        :subtitle="handleSubtitle(item)"
        @cancel="showAlert('cancel')"
        @remove="handleRemoveImage(index)"
        @download="handleDownload(index)"
        @click="handlePreview(item)"
      />
    </MpFlex>

    <MpModal :isOpen="modal.isOpen" @close="modal.isOpen = false" scrollBehavior="auto">
      <MpModalContent>
        <MpModalHeader>
          {{ modal.title }}

          <MpModalCloseButton />
        </MpModalHeader>
        <MpModalBody>
          <img :src="modal.content" alt="preview" style="border-radius: 6px" />
        </MpModalBody>
        <MpModalFooter>
          <MpButton variant="ghost"> Back </MpButton>
        </MpModalFooter>
      </MpModalContent>

      <MpModalOverlay />
    </MpModal>
  </div>
</template>

<script setup lang="ts">
  import { ref } from 'vue'
  import {
    MpUpload,
    MpUploadList,
    MpFlex,
    MpModal,
    MpModalOverlay,
    MpModalContent,
    MpModalHeader,
    MpModalFooter,
    MpModalBody,
    MpModalCloseButton,
    MpButton
  } from '@mekari/pixel3'

  const isLoading = ref(false)
  const isInvalid = ref(false)
  const images = ref()
  const modal = ref({
    isOpen: false,
    content: '',
    title: ''
  })

  function showAlert(message: string) {
    alert(message)
  }

  function handleStatus(item: File) {
    if (isLoading.value) {
      return 'loading'
    }

    if (item.size > 1000000) {
      return 'error'
    }

    return 'success'
  }

  function handleSubtitle(item: File) {
    if (isLoading.value) {
      return 'Uploading...'
    }

    if (item.size > 1000000) {
      return 'You reached maximum file size'
    }

    return `${Math.floor(item.size / 1000)}kb`
  }

  function handleChange(e: Event) {
    const target = e.target as HTMLInputElement

    images.value = target.files
    console.log('IMAGE: ', images.value)

    isLoading.value = true
    setTimeout(() => {
      // Upload image to server
      isLoading.value = false
    }, 1000)
  }

  function handleClear() {
    images.value = null
  }

  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  function handlePreview(item: any) {
    modal.value.isOpen = true
    modal.value.title = item.name
    modal.value.content = URL.createObjectURL(item)
  }

  function handleRemoveImage(index: number) {
    const dt = new DataTransfer()
    for (let i = 0; i < images.value.length; i++) {
      const file = images.value[i]
      if (index !== i) dt.items.add(file) // here you exclude the file. thus removing it.
    }
    images.value = dt.files
  }

  function handleDownload(index: number) {
    const file = images.value[index]
    const url = URL.createObjectURL(file)
    const a = document.createElement('a')
    a.href = url
    a.download = file.name
    document.body.appendChild(a)
    a.click()
    document.body.removeChild(a)
    URL.revokeObjectURL(url)
  }
</script>

```
